1. 创建版本库
	1.1 创建一个空目录
		$ mkdir learngit	%创建一个文件夹
		$ cd learngit		%进入某个文件夹（或用于切换进入某个文件夹，地址开头是/c/Users/...）
		$ pwd			%显示当前目录
	1.2 创建Git可以管理的仓库
		$ git init			%初始化一个Git仓库
	1.3 把文件添加到仓库
		$ git add readme.txt					%把文件添加到仓库暂存区（实际不用加文件后缀）
		$ git commit -m "wrote a readme file" 	%将文件从暂存区提交到仓库
2. 全局配置Git所使用的邮箱和昵称
	git config --global user.email "song971222@163.com"	%配置全局提交邮箱
	git config --global user.name "LINKS"				%配置全局提交用户名
3. 查看提交历史、当前文件状态、提交新版本
	3.1查看提交历史、当前文件状态
		$ git log			%查看提交历史
		$ git log --pretty=oneline --abbrev-commit	%以最简格式显示所有历史版本记录
		$ git status		%查看当前文件状态
		$ git diff Git教程.txt	%查看修改的内容（修改后的文件未提交仓库可用于查看修改的内容对比）
	3.2修改文件后，再次提交新版本
		$ git add .						%将当前文件下的所有修改都加入暂存区
		$ git commit -m "修改内容备注"		%提交新版本（把暂存区的内容提交到当前分支）
4. 版本回退
	4.1回退到某个版本状态（HEAD表示最新版本，HEAD^表示上一个版本，HEAD^~100表示往前100个版本）
		$ git reset --hard HEAD^	
（--hard参数有啥意义？--hard会回退到上个版本的已提交状态，而--soft会回退到上个版本的未提交状态，--mixed会回退到上个版本已添加但未提交的状态）
		$ git reset --hard XXXXX		
（XXX表示$git log历史记录中不同版本的版本号commit id，可以用于回到历史记录中的某个版本状态（跳转）。注意：只要记得不同版本的版本号commit id（版本号没必要写全，前几位就可以了，Git会自动去找），在Git中总是可以跳转到某个版本状态）
5. 撤销修改和删除文件
	5.1 撤销修改
		$ git checkout -- readme.txt	%把文件readme.txt在工作区的修改全部撤销（已经通过$ git add添加进暂存区的修改或进一步通过$git commit添加到仓库的修改将不会被该指令撤销）
	5.2 删除文件
		$ git rm test.txt						%删除文件test.txt
		$ git commit -m "确认删除文件test.txt"	%确认删除文件
		$ git checkout -- test.txt				%撤销误删除文件操作

6. 远程仓库（https://github.com）
	$ pwd		%显示当前文件地址
	$ cd /c/Users/song9/Desktop/其他文件/Git		%将当前文件地址切换到要上传到远程库的文件地址
	
	查看用户主目录下是否有.ssh目录，以及id-rsa和id-rsa_pub这两个文件；
	$ cat ~/.ssh/id-rsa.pub  %获取公钥id-rsa.pub；如果没有，需要先用$ ssh-keygen -t rsa -C "youremail@example.com"进行创建再查询
	登陆github.com，在账户设置中添加SSH Keys（命名随意）

	登陆github.com，选择/创建一个仓库repository（命名为LearningforGit）
	$ git remote add origin git@github.com:Logsti/LearningforGit.git 	%在本地仓库（Git文件作为当前地址）下，绑定远程仓库
	$ git push -u origin master	%第一次上传文件
	$ git push origin master		%后续再上传文件 

	$ git remote -v			%查看当前的远程库信息
	$ git remote rm origin	%解除与当前远程库的绑定

	$ git clone git@github.com:Logsti/LearningforGit.git		%从远程库中克隆文件到本地当前文件地址
7. 分支管理
	6.1 创建与合并分支
		$ git switch -c dev		%创建并切换到dev分支（create）
		$ git switch master		%切换到master分支
	6.2 合并分支
		$ git merge dev		%合并指定分支dev到当前分支
8. 标签管理
	8.1 创建标签
		$ git branch		%查看所有分支
		$ git switch master	%切换到master分支
		$ git tag v1.0		%打一个新标签
（默认标签是打在最新提交的commit上的，过往版本的标签需要先通过commit id回到之前的版本状态再进行打标签）
		$ git tag v0.9 f52c633	%给历史提交的commit id=f52c633打上新标签v0.9
		$ git tag			%查看所有标签
		$ git show v0.9		%查看标签信息
		$ git tag -a v0.9 -m "version 0.1 released" f52c633	%-a指定标签名，-m指定说明文字
	8.2 操作标签
		$ git tag -d v0.1		%删除标签（deleted）
		$ git push origin v0.1		%推送某个标签到远程
		$ git push origin --tags	%一次性推送全部尚未推送到远程的本地标签
		$ git push origin :refs/tags/v0.1		%删除远程标签，需要先删除本地标签$ git tag -d v0.1
